const getCustomerSelectBlock = async ({ airtableReqOptions }) => {
  const customersRes = await fetch(
    `${process.env.TABLE_BASE_URL}/formic_customers`,
    airtableReqOptions,
  );
  const customersData = await customersRes.json();
  console.log("customersData", customersData);
  let customersRecords = customersData.records;
  let offset = customersData.offset;

  while (offset) {
    const nextCustomersRes = await fetch(
      `${process.env.TABLE_BASE_URL}/formic_customers?offset=${offset}`,
      airtableReqOptions,
    );
    const nextCustomersData = await nextCustomersRes.json();
    console.log("nextCustomersData", nextCustomersData);

    const nextCustomersRecords = nextCustomersData.records;
    customersRecords = [...customersRecords, ...nextCustomersRecords];
    offset = nextCustomersData.offset;
  }

  const letterCategories = "ABCDEFGHIJKLMNOPQRSTUVWXYZ".split("");
  const numericCategories = "#";
  const sections = [numericCategories, ...letterCategories];

  const sortedCustomersRecords = customersRecords.sort((a, b) =>
    a.fields.name.localeCompare(b.fields.name),
  );

  const sectionedSortedCustomersRecords = sections.reduce((acc, curr) => {
    if (curr === "#") {
      const filteredRecords = sortedCustomersRecords.filter((record) =>
        parseInt(record.fields.name[0]),
      );
      if (filteredRecords.length > 0) {
        const options = filteredRecords.map((record) => {
          return createOption(record.fields.name, record.id);
        });
        const optionGroup = {
          label: {
            type: "plain_text",
            text: curr,
          },
          options: options,
        };
        acc.push(optionGroup);
      }
    } else {
      const filteredRecords = sortedCustomersRecords.filter(
        (record) => record.fields.name[0] === curr,
      );
      if (filteredRecords.length > 0) {
        const options = filteredRecords.map((record) => {
          return createOption(record.fields.name, record.id);
        });
        const optionGroup = {
          label: {
            type: "plain_text",
            text: curr,
          },
          options: options,
        };
        acc.push(optionGroup);
      }
    }
    return acc;
  }, []);
  const customerOptions = sortedCustomersRecords.map((record) =>
    createOption(record.fields.name, record.id),
  );

  const customerSelectBlock = createSelectBlock({
    label: "Which customer?",
    blockId: "customer",
    actionId: "customer",
    optionGroups: sectionedSortedCustomersRecords,
    optional: false,
    dispatchAction: true,
  });
  return customerSelectBlock;
};

const flattenSlackViewState = (state) => {
  const result = {};
  const values = state.values;
  const blockKeys = Object.keys(values);
  for (const blockKey of blockKeys) {
    const actionKeys = Object.keys(values[blockKey]);
    for (const actionKey of actionKeys) {
      const type = values[blockKey][actionKey].type;
      switch (type) {
        case "plain_text_input":
          result[actionKey] = values[blockKey][actionKey].value;
          break;
        case "email_text_input":
          result[actionKey] = values[blockKey][actionKey].value;
          break;
        case "number_input":
          result[actionKey] = values[blockKey][actionKey].value;
          break;
        case "radio_buttons":
          result[actionKey] = values[blockKey][actionKey].selected_option.value;
          break;
        case "static_select":
          if (
            actionKey === "customer" ||
            actionKey === "contact" ||
            actionKey === "site"
          ) {
            result[actionKey] = [
              values[blockKey][actionKey].selected_option.value,
            ];
            break;
          } else if (actionKey === "asa") {
            result[actionKey] = [
              values[blockKey][actionKey].selected_option.value,
            ];
            result["system_name"] =
              values[blockKey][actionKey].selected_option.text.text;
            break;
          }
          result[actionKey] = values[blockKey][actionKey].selected_option.value;
          break;
        case "users_select":
          result[actionKey] = values[blockKey][actionKey].selected_user;
          break;
        case "multi_users_select":
          result[actionKey] =
            values[blockKey][actionKey].selected_users.join(",");
          break;
        case "file_input":
          result[actionKey] = values[blockKey][actionKey].files
            .map((file) => file.permalink)
            .join(",");
        default:
          break;
      }
    }
  }
  return result;
};

const onSubmitTicket = async ({
  $,
  payload,
  airtableReqOptions,
  webClient,
}) => {
  await $.respond({
    status: "200",
    body: {
      response_action: "clear",
    },
  });

  const state = payload.view.state;
  const submitterId = payload.user.id;
  const flattenedViewState = flattenSlackViewState(state);
  console.log("flattenedViewState", flattenedViewState);
  const { system_name, ...viewStateForSubmit } = flattenedViewState;
  const assigneeInfo = await webClient.users.info({
    user: flattenedViewState.assignee,
  });
  const notifyInfo = flattenedViewState.notify
    ? await webClient.users.info({ user: flattenedViewState.notify })
    : "";
  const submitterInfo = await webClient.users.info({ user: submitterId });

  const submitFields = {
    ...viewStateForSubmit,
    assignee_name: assigneeInfo.user.real_name,
    submitter_name: submitterInfo.user.real_name,
    submitter: submitterId,
  };

  if (notifyInfo?.user?.real_name) {
    submitFields.notify_name = notifyInfo.user.real_name;
  }

  const submitPayload = JSON.stringify({ fields: submitFields });
  console.log("submitPayload", submitPayload);

  const submit = await fetch(
    `${process.env.TABLE_BASE_URL}/inboundescalations_2`,
    {
      method: "POST",
      body: submitPayload,
      ...airtableReqOptions,
    },
  );
  const submitInfo = await submit.json();
  console.log("submitInfo", submitInfo);

  const recordId = submitInfo.fields.record_id;
  const updateFields = { record_id_slack: recordId };
  const updatePayload = JSON.stringify({ fields: updateFields });
  console.log(updatePayload);
  const update = await fetch(
    `${process.env.TABLE_BASE_URL}/inboundescalations_2/${recordId}`,
    {
      method: "PATCH",
      body: updatePayload,
      ...airtableReqOptions,
    },
  );
  const updateInfo = await update.json();
  console.log("update", updateInfo);

  const webhookBody = {
    contact_phone_number: submitInfo.fields.contact_phone_number[0],
    record_id: submitInfo.fields.record_id,
    ticket_id: submitInfo.fields.ticket_id,
    description: submitInfo.fields.description,
    assignee: submitInfo.fields.assignee,
    customer_name: submitInfo.fields.customer_name[0],
    submitter: submitInfo.fields.submitter,
    priority: submitInfo.fields.priority,
    system_id: submitInfo.fields.asa[0],
    system_name: flattenedViewState.system_name,
    cs_channel:
      process.env.ENV === "TEST"
        ? "C0A68D2UN57"
        : submitInfo.fields.cs_channel[0],
    contact_name: submitInfo.fields.contact_full_name[0],
    comms_direction: submitInfo.fields.comms_direction,
    follow_up: submitInfo.fields.follow_up,
  };
  if (submitInfo.fields.notify) {
    webhookBody.notify = submitInfo.fields.notify;
  }
  switch (submitInfo.fields.priority) {
    case PRIORITY["P5"]:
      webhookBody.main_channel =
        process.env.CUSTOMER_SUCCESS_ACTIVITY_CHANNEL_ID;
      break;
    case PRIORITY["P6"]:
      webhookBody.main_channel =
        process.env.ANT_CONNECTIVITY_DATA_ISSUES_CHANNEL_ID;
      break;
    default:
      webhookBody.main_channel =
        process.env.INBOUND_SUPPORT_ESCALATIONS_CHANNEL_ID;
      break;
  }
  console.log("webhookBody", webhookBody);

  const webhookUrl =
    "https://hooks.slack.com/triggers/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";
  const webhookUrlNotify =
    "https://hooks.slack.com/triggers/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";

  const hitWebhook = async (url, body) =>
    await fetch(url, {
      headers: {
        "Content-Type": "application/json",
      },
      method: "POST",
      body: JSON.stringify(body),
    });
  const webhookStream = await hitWebhook(
    submitInfo.fields.notify ? webhookUrlNotify : webhookUrl,
    webhookBody,
  );
  const webhookJson = await webhookStream.json();
  console.log("webhookJson", webhookJson);
};

const onSubmitContact = async ({ $, webClient, view, airtableReqOptions }) => {
  const state = view.state;
  const private_metadata = view.private_metadata;
  const privateMetadata = JSON.parse(private_metadata);
  const selectedCustomerId = privateMetadata.selectedCustomerId;
  const selectedCustomerName = privateMetadata.selectedCustomerName;
  const viewId = privateMetadata.viewId;

  const fields = {
    ...flattenSlackViewState(state),
    customer: [selectedCustomerId],
  };
  const payload = JSON.stringify({
    fields: fields,
  });
  const submit = await fetch(
    `${process.env.TABLE_BASE_URL}/formic_customer_contacts`,
    {
      method: "POST",
      body: payload,
      ...airtableReqOptions,
    },
  );
  const submitInfo = await submit.json();
  console.log("submitinfo", submitInfo);
  await $.respond({
    status: "200",
  });

  const customerSelectBlock = await getCustomerSelectBlock({
    airtableReqOptions,
  });
  const ticketFormBlocks = await getTicketFormBlocks({
    airtableReqOptions,
    selectedCustomerName,
  });
  const blocks = [customerSelectBlock, ...ticketFormBlocks];

  const newView = {
    view_id: viewId,
    view: {
      type: "modal",
      callback_id: "ticket_form",
      title: {
        type: "plain_text",
        text: "Inbound support request",
      },
      blocks,
      submit: {
        type: "plain_text",
        text: "Submit",
      },
      close: {
        type: "plain_text",
        text: "Cancel",
      },
    },
  };

  await webClient.views.update(newView);
  await $.respond({
    status: "200",
  });
};

export default defineComponent({
  async run({ steps, $ }) {
    const payload = JSON.parse(steps.trigger.event.body.payload);
    const airtableReqOptions = {
      headers: {
        Authorization: process.env.AIRTABLE_TOKEN,
        "Content-Type": "application/json",
      },
    };
    const token = process.env.SLACK_TOKEN;
    const webClient = new WebClient(token);
    const triggerId = payload.trigger_id;

    if (payload.type === "view_submission") {
      const callbackId = payload.view.callback_id;
      switch (callbackId) {
        case "ticket_form":
          await onSubmitTicket({
            $,
            payload,
            airtableReqOptions,
            webClient,
          });
          break;
        case "contact_form":
          await onSubmitContact({
            $,
            view: payload.view,
            airtableReqOptions,
            webClient,
          });
          break;
        default:
          break;
      }
      return;
    } else if (payload.type === "block_actions") {
      await $.respond({
        status: "200",
      });
      const viewId = payload.view.id;
      for (const action of payload.actions) {
        const payloadBlocks = payload.view.blocks;
        switch (action.action_id) {
          case "customer":
            const selectedCustomerOption = action.selected_option;
            await onCustomerSelect({
              $,
              airtableReqOptions,
              webClient,
              viewId,
              selectedCustomerName: selectedCustomerOption.text.text,
              selectedCustomerId: selectedCustomerOption.value,
              existingBlocks: payloadBlocks.slice(0, 1),
            });
            break;
          case "add-contact":
            await onAddContact({
              $,
              airtableReqOptions,
              webClient,
              viewId,
              triggerId,
              state: payload.view.state,
            });
            break;
          default:
            break;
        }
      }
      return;
    } else {
      await $.respond({
        status: "200",
      });
      await onOpenModal({ $, airtableReqOptions, webClient, triggerId });
    }
  },
});
